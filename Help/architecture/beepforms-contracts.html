Overview

Namespace: TheTechIdea.Beep.Winform.Controls.Integrated.Forms.Contracts
Two interfaces define the contract between BeepBlock and BeepForms (the Coordinator host). The architecture follows a strict discipline:

  • IBeepFormsHost -- A facade that BeepBlock calls. Blocks must never dereference FormsManager directly; they must use only the proxy methods on this interface. This keeps the block-layer decoupled from the forms infrastructure.
  • IBeepFormsBootstrapper -- Initializes blocks with EntityStructure metadata from live datasources.
BeepBlock -> IBeepFormsHost (BeepForms) -> IUnitofWorksManager (FormsManager) -> IDataSource
   ^--- Block must ONLY call IBeepFormsHost, never dereference FormsManager

IBeepFormsBootstrapper

public interface IBeepFormsBootstrapper
{
    Task<bool> BootstrapAsync(BeepForms forms,
        CancellationToken cancellationToken = default);
    // Populates BeepBlock instances with EntityStructure metadata
    // from a live datasource. Iterates all blocks in the
    // BeepFormsDefinition with valid ConnectionName + EntityName.
    //
    // Returns: true = all blocks populated,
    //          false = at least one block failed or was skipped.
    // Implementations must not throw; swallow failures.
}

Code Examples

// Block registration
var host = (IBeepFormsHost)myBeepForms;
host.RegisterBlock(myBlockView);
host.TrySetActiveBlock("CustomerBlock");

// Query current state
var uow = host.GetBlockUnitOfWork("CustomerBlock");
var currentItem = host.GetCurrentBlockItem("CustomerBlock");

// LOV interaction
if (host.HasLov("CustomerBlock", "Country"))
{
    var result = await host.ShowLovAsync(
        "CustomerBlock", "Country");
    if (result.Selected)
        Console.WriteLine($"Selected: {result.Value}");
}

// Save and validate
var validation = host.ValidateBlockRecord("CustomerBlock",
    myRecord, ValidationTiming.BeforeSave);
if (validation.IsValid)
    await host.SaveBlockAsync("CustomerBlock");

// Bootstrap
var bootstrapper = new MyBootstrapper();
await bootstrapper.BootstrapAsync(myBeepForms);
