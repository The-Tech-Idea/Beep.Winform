Overview

The Beep theme system powers the visual appearance of every control. It consists of:

  • IBeepTheme — Interface defining the theme contract (in TheTechIdea.Beep.Vis.Modules2.0)
  • BeepTheme — Central partial class with 35+ partial files, each handling a control type
  • Theme Types — Concrete implementations: Ubuntu, GNOME, Cyberpunk, Candy, Zen
  • BeepThemesManager — Singleton manager for theme registration and switching

IBeepTheme Contract

public interface IBeepTheme
{
    string ThemeName { get; }
    Color ButtonBackColor { get; }
    Color ButtonForeColor { get; }
    Color ButtonBorderColor { get; }
    Color ButtonHoverColor { get; }
    // ... 200+ color/style properties for every control type
    Font DefaultFont { get; }
    Font TitleFont { get; }
    int BorderRadius { get; }
    float Opacity { get; }
}

Token Resolution

// Token resolution pipeline:
// 1. Control checks its own theme property override
// 2. Falls back to BeepThemesManager.CurrentTheme
// 3. Falls back to default theme values

public Color GetButtonBackColor(BaseControl control)
{
    if (control.ThemeOverrides?.ButtonBackColor != null)
        return control.ThemeOverrides.ButtonBackColor;

    return BeepThemesManager.CurrentTheme
        ?.ButtonBackColor ?? DefaultTheme.ButtonBackColor;
}

Examples

// Global theme switch
BeepThemesManager.Instance.SetTheme("DarkTheme");
// All controls update automatically

// Per-control override
myButton.ThemeOverrides = new ThemeOverrides
{
    ButtonBackColor = Color.Red,
    BorderRadius = 12
};

// Custom theme
var myTheme = new BeepTheme
{
    ThemeName = "MyCustomTheme",
    ButtonBackColor = Color.Navy,
    ButtonForeColor = Color.White,
    // ... configure all properties
};
BeepThemesManager.Instance.RegisterTheme(myTheme);
