Overview

BeepPanel is an enhanced container control that extends the standard Windows Forms Panel with optional title bars, improved theming support, border customization, and flexible layout options for organizing content in your applications.

Built on the BeepControl foundation, BeepPanel provides a clean, modern container that can display titles, icons, and action buttons while maintaining seamless integration with the Beep theming system.

Basic Usage

Simple Panel

// Create a basic panel
var panel = new BeepPanel
{
    Size = new Size(300, 200),
    Location = new Point(10, 10),
    BorderStyle = BorderStyle.FixedSingle
};

// Add some content
var label = new Label
{
    Text = "Panel Content",
    Dock = DockStyle.Fill,
    TextAlign = ContentAlignment.MiddleCenter
};

panel.Controls.Add(label);
this.Controls.Add(panel);

Panel with Title Bar

// Create panel with title bar
var titlePanel = new BeepPanel
{
    ShowTitle = true,

    BorderRadius = 8

// Add content to the panel
var contentArea = new BeepPanel

    AutoSize = true

üí° Layout Tip

When using title bars, the content area automatically adjusts to fill the remaining space. Use Dock = DockStyle.Fill for child controls to take advantage of this.

Title Bar

The title bar provides a clean header area for your panel with support for text, icons, and action buttons.

Title Bar with Action Buttons

var panel = new BeepPanel
{
    ShowTitle = true,
    TitleText = "Data Panel",
    TitleIcon = "üìä", // Emoji or use SVG
    ShowMinimizeButton = true,
    ShowMaximizeButton = true,
    ShowCloseButton = true,
    TitleHeight = 35,
    Size = new Size(500, 400),
    Dock = DockStyle.Fill
};

// Handle title bar button events
panel.MinimizeClicked += (s, e) => {
    panel.Height = panel.TitleHeight;
    panel.Minimized = true;
};

panel.MaximizeClicked += (s, e) => {
    if (panel.WindowState == FormWindowState.Maximized)
    {
        panel.WindowState = FormWindowState.Normal;
    }
    else
    {
        panel.WindowState = FormWindowState.Maximized;
    }
};

panel.CloseClicked += (s, e) => {
    panel.Visible = false;
    // Or remove from parent container
    // panel.Parent.Controls.Remove(panel);
};

Examples

Dashboard Layout with Multiple Panels

private void CreateDashboardLayout()
{
    var mainContainer = new BeepPanel
    {
        Dock = DockStyle.Fill,
        Padding = new Padding(10)
    };

    // Header panel
    var headerPanel = new BeepPanel
    {
        ShowTitle = true,
        TitleText = "Dashboard Overview",
        TitleIcon = "üìä",
        Dock = DockStyle.Top,
        Height = 80,
        Margin = new Padding(0, 0, 0, 10),
        BorderRadius = 8
    };

    // Left sidebar panel
    var sidebarPanel = new BeepPanel
    {
        ShowTitle = true,
        TitleText = "Navigation",
        Dock = DockStyle.Left,
        Width = 200,
        Margin = new Padding(0, 0, 10, 0),
        BorderRadius = 8
    };

    // Main content panel
    var contentPanel = new BeepPanel
    {
        ShowTitle = true,
        TitleText = "Main Content",
        TitleIcon = "üìã",
        ShowMaximizeButton = true,
        Dock = DockStyle.Fill,
        BorderRadius = 8
    };

    // Add navigation items to sidebar
    var navItems = new string[] { "Home", "Reports", "Settings", "Help" };
    var yPosition = 10;

    foreach (var item in navItems)
    {
        var navButton = new BeepButton
        {
            Text = item,
            Size = new Size(180, 35),
            Location = new Point(10, yPosition),
            UIVariant = ReactUIVariant.Text,
            UIColor = ReactUIColor.Primary
        };

        sidebarPanel.Controls.Add(navButton);
        yPosition += 45;
    }

    mainContainer.Controls.AddRange(new Control[]
    {
        headerPanel, sidebarPanel, contentPanel
    });

    this.Controls.Add(mainContainer);
}

Collapsible Settings Panel

private void CreateCollapsiblePanel()
{
    var settingsPanel = new BeepPanel
    {
        ShowTitle = true,
        TitleText = "Advanced Settings",
        TitleIcon = "‚öôÔ∏è",
        ShowMinimizeButton = true,
        Size = new Size(400, 300),
        Location = new Point(20, 20),
        BorderRadius = 10
    };

    // Create content that can be collapsed
    var contentPanel = new BeepPanel
    {
        Dock = DockStyle.Fill,
        Padding = new Padding(15),
        AutoScroll = true
    };

    // Add settings controls
    var settings = new Control[]
    {
        new BeepCheckBox { Text = "Enable auto-save", Location = new Point(0, 0), AutoSize = true },
        new BeepCheckBox { Text = "Show notifications", Location = new Point(0, 30), AutoSize = true },
        new BeepCheckBox { Text = "Dark mode", Location = new Point(0, 60), AutoSize = true },
        new BeepComboBox {
            DropDownStyle = ComboBoxStyle.DropDownList,
            Location = new Point(0, 90),
            Size = new Size(200, 25)
        }
    };

    contentPanel.Controls.AddRange(settings);
    settingsPanel.Controls.Add(contentPanel);

    // Handle collapse/expand
    bool isCollapsed = false;
    int originalHeight = settingsPanel.Height;

    settingsPanel.MinimizeClicked += (s, e) => {
        if (isCollapsed)
        {
            // Expand
            settingsPanel.Height = originalHeight;
            contentPanel.Visible = true;
            isCollapsed = false;
        }
        else
        {
            // Collapse
            originalHeight = settingsPanel.Height;
            settingsPanel.Height = settingsPanel.TitleHeight;
            contentPanel.Visible = false;
            isCollapsed = true;
        }
    };

    this.Controls.Add(settingsPanel);
}

Theming

BeepPanel integrates seamlessly with the Beep theming system, automatically adapting colors, fonts, and visual effects.

Applying Themes

// Apply theme to individual panel
panel.Theme = "DarkTheme";
panel.ApplyTheme();

// Apply theme to all panels recursively
private void ApplyThemeToPanels(Control container, string themeName)
{
    foreach (Control control in container.Controls)
    {
        if (control is BeepPanel panel)
        {
            panel.Theme = themeName;
            panel.ApplyTheme();
        }

        // Recursively apply to nested containers
        if (control.HasChildren)
        {
            ApplyThemeToPanels(control, themeName);
        }
    }
}

// Apply to entire form
ApplyThemeToPanels(this, "DarkTheme");

Theme-Responsive Properties

  • Background colors - Automatically use theme background colors
  • Border colors - Adapt to theme accent and border colors
  • Title bar styling - Use theme colors for title text and buttons
  • Shadow effects - Adjust shadow colors based on theme

Best Practices

üé® Visual Design

  • Consistent spacing: Use consistent margins and padding across panels
  • Logical grouping: Group related content within panels
  • Visual hierarchy: Use title bars to establish content importance
  • Appropriate sizing: Size panels based on their content and context

‚ö° Performance

  • Minimize nesting: Avoid excessive panel nesting for better performance
  • Use AutoScroll wisely: Enable only when necessary to avoid layout issues
  • Batch updates: Suspend layout when adding multiple controls

‚ôø Accessibility

  • Meaningful titles: Use descriptive title text for screen readers
  • Keyboard navigation: Ensure panel contents are keyboard accessible
  • High contrast support: Test panels with high contrast themes
  • Focus management: Handle focus properly when collapsing/expanding
‚ö†Ô∏è Layout Considerations

When using title bars, remember that they consume vertical space. Plan your layout accordingly and consider using shorter titles for small panels.

‚úÖ Pro Tip

Use the Dock property strategically with BeepPanel to create responsive layouts that adapt to window resizing automatically.
