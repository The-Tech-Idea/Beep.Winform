?? Overview

BeepThemesManager is the central theming system for all Beep Controls. It provides enterprise-grade theme management with dynamic loading, persistence, and comprehensive customization capabilities. The system supports 32+ built-in themes and enables seamless theme switching across entire applications.

Built as a static manager class, BeepThemesManager automatically discovers and loads theme implementations from assemblies, provides event-driven theme changes, and integrates with the BeepFontManager for comprehensive typography management.

?? Key Features

?? 32+ Built-in Themes

Comprehensive collection from Default to specialized themes like CyberpunkNeon and GlassMorphism

?? Dynamic Loading

Automatic discovery and loading of theme implementations from assemblies

Real-time theme change notifications with complete event system

Save and load custom themes with XML serialization

Seamless integration with BeepFontManager for comprehensive font management

Thread-safe operations, GUID-based identification, and robust error handling

?? Basic Usage

Getting and Setting Themes

// Get current theme
IBeepTheme currentTheme = BeepThemesManager.CurrentTheme;
string currentThemeName = BeepThemesManager.CurrentThemeName;

Console.WriteLine($"Current theme: {currentThemeName}");

// Set theme by name
BeepThemesManager.SetCurrentTheme("DarkTheme");

// Set theme by object
IBeepTheme oceanTheme = BeepThemesManager.GetTheme("OceanTheme");
BeepThemesManager.SetCurrentTheme(oceanTheme);

// Get specific theme
IBeepTheme materialTheme = BeepThemesManager.GetTheme("MaterialDesignTheme");

// Check if theme exists
if (BeepThemesManager.ThemeExists("CyberpunkNeonTheme"))
{
    BeepThemesManager.SetCurrentTheme("CyberpunkNeonTheme");
}

Using with Controls

// Apply theme to a control
var button = new BeepButton();
button.Theme = BeepThemesManager.CurrentThemeName;
button.ApplyTheme();

// Listen for theme changes
BeepThemesManager.ThemeChanged += (sender, e) => {
    Console.WriteLine($"Theme changed from {e.OldThemeName} to {e.NewThemeName}");
    
    // Update all controls
    button.Theme = e.NewThemeName;
    button.ApplyTheme();
};

// Set theme and all subscribed controls will update automatically
BeepThemesManager.SetCurrentTheme("MaterialDesignTheme");
?? Quick Start Tip

Use BeepThemesManager.CurrentTheme to get the active theme object and BeepThemesManager.CurrentThemeName for the theme name string.

Application Settings Integration

public class ThemeSettings
{
    public static void SaveCurrentTheme()
    {
        // Save current theme name to application settings
        Properties.Settings.Default.CurrentTheme = BeepThemesManager.CurrentThemeName;
        Properties.Settings.Default.Save();
    }
    
    public static void LoadSavedTheme()
    {
        // Load theme from application settings
        string savedTheme = Properties.Settings.Default.CurrentTheme;
        
        if (!string.IsNullOrEmpty(savedTheme) && BeepThemesManager.ThemeExists(savedTheme))
        {
            BeepThemesManager.SetCurrentTheme(savedTheme);
        }
        else
        {
            // Fallback to default
            BeepThemesManager.SetCurrentTheme("DefaultTheme");
        }
    }
    
    public static void ExportCurrentTheme(string filePath)
    {
        var currentTheme = BeepThemesManager.CurrentTheme;
        BeepThemesManager.SaveTheme(currentTheme, filePath);
    }
    
    public static void ImportTheme(string filePath)
    {
        var importedTheme = BeepThemesManager.LoadTheme(filePath);
        BeepThemesManager.SetCurrentTheme(importedTheme.ThemeName);
    }
}

Theme Backup and Restore

public class ThemeBackupManager
{
    private static readonly string BackupDirectory = @"C:\MyApp\ThemeBackups";
    
    public static void BackupAllCustomThemes()
    {
        Directory.CreateDirectory(BackupDirectory);
        
        var allThemes = BeepThemesManager.GetThemes();
        var customThemes = allThemes.Where(t => !IsBuiltInTheme(t.ThemeName));
        
        foreach (var theme in customThemes)
        {
            string fileName = $"{theme.ThemeName}_{DateTime.Now:yyyyMMdd}.xml";
            string filePath = Path.Combine(BackupDirectory, fileName);
            
            BeepThemesManager.SaveTheme(theme, filePath);
            Console.WriteLine($"Backed up theme: {theme.ThemeName}");
        }
    }
    
    public static void RestoreThemesFromBackup()
    {
        if (!Directory.Exists(BackupDirectory))
            return;
            
        var themeFiles = Directory.GetFiles(BackupDirectory, "*.xml");
        
        foreach (var file in themeFiles)
        {
            try
            {
                var theme = BeepThemesManager.LoadTheme(file);
                Console.WriteLine($"Restored theme: {theme.ThemeName}");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Failed to restore theme from {file}: {ex.Message}");
            }
        }
    }
    
    private static bool IsBuiltInTheme(string themeName)
    {
        var builtInThemes = new[] {
            "DefaultTheme", "DarkTheme", "LightTheme", "MaterialDesignTheme",
            "NeumorphismTheme", "CyberpunkNeonTheme", "OceanTheme"
            // ... add other built-in theme names
        };
        
        return builtInThemes.Contains(themeName);
    }
}

?? Event System

Theme Change Events

// Subscribe to theme change events
BeepThemesManager.ThemeChanged += OnThemeChanged;

private static void OnThemeChanged(object sender, ThemeChangeEventArgs e)
{
    Console.WriteLine($"Theme changed from {e.OldThemeName} to {e.NewThemeName}");
    
    // Access old and new theme objects
    IBeepTheme oldTheme = e.OldTheme;
    IBeepTheme newTheme = e.NewTheme;
    
    // Update application-wide UI
    UpdateApplicationUI(newTheme);
    
    // Save new theme preference
    SaveUserThemePreference(e.NewThemeName);
    
    // Notify other components
    NotifyComponentsOfThemeChange(e);
}

// Unsubscribe when no longer needed
BeepThemesManager.ThemeChanged -= OnThemeChanged;

Application-Wide Theme Management

public class ApplicationThemeManager
{
    private static List _themeAwareComponents = new List();
    
    public static void Initialize()
    {
        BeepThemesManager.ThemeChanged += OnApplicationThemeChanged;
    }
    
    public static void RegisterThemeAwareComponent(IThemeAware component)
    {
        _themeAwareComponents.Add(component);
        
        // Apply current theme immediately
        component.ApplyTheme(BeepThemesManager.CurrentTheme);
    }
    
    public static void UnregisterThemeAwareComponent(IThemeAware component)
    {
        _themeAwareComponents.Remove(component);
    }
    
    private static void OnApplicationThemeChanged(object sender, ThemeChangeEventArgs e)
    {
        // Update all registered components
        foreach (var component in _themeAwareComponents.ToList())
        {
            try
            {
                component.ApplyTheme(e.NewTheme);
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Error applying theme to component: {ex.Message}");
                
                // Remove failed components
                _themeAwareComponents.Remove(component);
            }
        }
        
        // Update system colors if needed
        UpdateSystemColors(e.NewTheme);
        
        // Refresh main windows
        RefreshAllWindows();
    }
    
    public static void ChangeTheme(string themeName)
    {
        if (BeepThemesManager.ThemeExists(themeName))
        {
            BeepThemesManager.SetCurrentTheme(themeName);
        }
        else
        {
            throw new ArgumentException($"Theme '{themeName}' not found");
        }
    }
}

Production-Ready Theme Manager

public class ProductionThemeManager
{
    private static readonly ILogger Logger = LogManager.GetCurrentClassLogger();
    
    public static void InitializeForProduction()
    {
        try
        {
            // Load user's preferred theme
            LoadUserPreferredTheme();
            
            // Set up automatic theme persistence
            BeepThemesManager.ThemeChanged += SaveThemePreference;
            
            // Validate all themes
            ValidateAllThemes();
            
            Logger.Info($"Theme system initialized with {BeepThemesManager.GetThemes().Count} themes");
        }
        catch (Exception ex)
        {
            Logger.Error(ex, "Failed to initialize theme system");
            
            // Fallback to default theme
            BeepThemesManager.SetCurrentTheme("DefaultTheme");
        }
    }
    
    private static void ValidateAllThemes()
    {
        var themes = BeepThemesManager.GetThemes();
        var invalidThemes = new List();
        
        foreach (var theme in themes)
        {
            if (!ValidateTheme(theme))
            {
                invalidThemes.Add(theme.ThemeName);
                Logger.Warn($"Theme '{theme.ThemeName}' failed validation");
            }
        }
        
        // Remove invalid themes
        foreach (var themeName in invalidThemes)
        {
            var theme = BeepThemesManager.GetTheme(themeName);
            BeepThemesManager.RemoveTheme(theme);
        }
    }
    
    private static bool ValidateTheme(IBeepTheme theme)
    {
        // Validate required properties
        if (string.IsNullOrEmpty(theme.ThemeName) || 
            string.IsNullOrEmpty(theme.ThemeGuid))
            return false;
            
        // Validate color contrast
        if (!HasAdequateContrast(theme.BackColor, theme.ForeColor))
            return false;
            
        // Add more validation rules as needed
        return true;
    }
    
    private static bool HasAdequateContrast(Color background, Color foreground)
    {
        // Implement WCAG contrast ratio calculation
        double contrastRatio = CalculateContrastRatio(background, foreground);
        return contrastRatio >= 4.5; // WCAG AA standard
    }
}
