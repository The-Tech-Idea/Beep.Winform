Overview

BeepCompanyProfile provides a comprehensive, professional display component for showcasing company or organization information. It combines visual branding elements with detailed contact information, key metrics, and social media integration in a cohesive, branded interface.

Built on the BeepControl foundation, BeepCompanyProfile features flexible layout options, customizable branding elements, metric displays, social media integration, contact information management, and comprehensive theming support. Perfect for about pages, company dashboards, CRM systems, and any application requiring professional company presentation.

Basic Usage

Simple Company Profile

// Create and configure a BeepCompanyProfile
var companyProfile = new BeepCompanyProfile();
companyProfile.Size = new Size(400, 300);
companyProfile.CompanyName = "TheTechIdea";
companyProfile.Tagline = "Innovative Software Solutions";
companyProfile.Description = "Leading provider of enterprise software solutions.";
companyProfile.Logo = Properties.Resources.CompanyLogo;

// Set contact information
companyProfile.Website = "https://thetechidea.net";
companyProfile.Email = "info@thetechidea.net";
companyProfile.Phone = "+1 (555) 123-4567";
companyProfile.Address = "San Francisco, CA";

// Handle profile interactions
companyProfile.ContactClicked += (s, e) =>
{
    switch (e.ContactType)
    {
        case ContactType.Email:
            Process.Start($"mailto:{companyProfile.Email}");
            break;
        case ContactType.Website:
            Process.Start(companyProfile.Website);
            break;
        case ContactType.Phone:
            MessageBox.Show($"Call {companyProfile.Phone}");
            break;
    }
};

// Add to form or panel
this.Controls.Add(companyProfile);

Profile with Metrics and Social Media

// Create comprehensive company profile
var profile = new BeepCompanyProfile
{
    CompanyName = "TheTechIdea",
    Tagline = "Innovative Software Solutions",
    Description = "We help organizations transform through technology",
    Size = new Size(500, 600),

// Add key metrics
profile.AddMetric("Founded", "2020");
profile.AddMetric("Employees", "50+");
profile.AddMetric("Projects", "200+");
profile.AddMetric("Countries", "25+");

// Add social media links

profile.AddCertification("Microsoft Partner", CertificationType.Technology);
profile.AddCertification("AWS Certified", CertificationType.Cloud);

// Handle social media clicks
profile.SocialLinkClicked += (s, e) =>
{
    Process.Start(e.Url);
    LogSocialInteraction(e.Platform, e.Url);
};

Examples

Corporate Dashboard Integration

public partial class CorporateDashboard : UserControl
{
    private BeepCompanyProfile companyProfile;
    private ICompanyService companyService;
    private IAnalyticsService analyticsService;

    public CorporateDashboard(ICompanyService companyService, IAnalyticsService analyticsService)
    {
        this.companyService = companyService;
        this.analyticsService = analyticsService;
        InitializeComponents();
        LoadCompanyProfile();
    }

    private void InitializeComponents()
    {
        companyProfile = new BeepCompanyProfile
        {
            Dock = DockStyle.Top,
            Height = 400,
            ProfileLayout = CompanyProfileLayout.Detailed,
            ShowMetrics = true,
            ShowSocialLinks = true,
            ShowCertifications = true
        };

        // Configure event handlers
        companyProfile.ContactClicked += OnContactClicked;
        companyProfile.SocialLinkClicked += OnSocialLinkClicked;
        companyProfile.MetricClicked += OnMetricClicked;
        companyProfile.LogoClicked += OnLogoClicked;

        this.Controls.Add(companyProfile);
    }

    private async void LoadCompanyProfile()
    {
        try
        {
            var companyInfo = await companyService.GetCompanyInfoAsync();
            var metrics = await analyticsService.GetCompanyMetricsAsync();

            // Set basic company information
            companyProfile.CompanyName = companyInfo.Name;
            companyProfile.Tagline = companyInfo.Tagline;
            companyProfile.Description = companyInfo.Description;
            companyProfile.Logo = await LoadImageAsync(companyInfo.LogoUrl);

            // Set contact information
            companyProfile.Website = companyInfo.Website;
            companyProfile.Email = companyInfo.ContactEmail;
            companyProfile.Phone = companyInfo.ContactPhone;
            companyProfile.Address = companyInfo.Address;

            // Add dynamic metrics
            foreach (var metric in metrics)
            {
                companyProfile.AddMetric(metric.Name, metric.Value, metric.Icon);
            }

            // Add social media links
            foreach (var socialLink in companyInfo.SocialLinks)
            {
                companyProfile.AddSocialLink(
                    socialLink.Name,
                    socialLink.Url,
                    ParseSocialPlatform(socialLink.Platform)
                );
            }

            // Add certifications
            foreach (var cert in companyInfo.Certifications)
            {
                companyProfile.AddCertification(
                    cert.Name,
                    cert.Type,
                    cert.ExpiryDate,
                    cert.BadgeImage
                );
            }
        }
        catch (Exception ex)
        {
            MessageBox.Show($"Error loading company profile: {ex.Message}");
        }
    }

    private async void OnContactClicked(object sender, ContactEventArgs e)
    {
        try
        {
            // Track contact interaction
            await analyticsService.TrackContactInteractionAsync(e.ContactType);

            switch (e.ContactType)
            {
                case ContactType.Email:
                    if (IsEmailClientAvailable())
                    {
                        Process.Start($"mailto:{e.ContactValue}");
                    }
                    else
                    {
                        Clipboard.SetText(e.ContactValue);
                        ShowNotification($"Email address copied to clipboard: {e.ContactValue}");
                    }
                    break;

                case ContactType.Phone:
                    if (IsDialerAvailable())
                    {
                        Process.Start($"tel:{e.ContactValue}");
                    }
                    else
                    {
                        Clipboard.SetText(e.ContactValue);
                        ShowNotification($"Phone number copied to clipboard: {e.ContactValue}");
                    }
                    break;

                case ContactType.Website:
                    Process.Start(e.ContactValue);
                    break;

                case ContactType.Address:
                    var mapsUrl = $"https://maps.google.com/maps?q={Uri.EscapeDataString(e.ContactValue)}";
                    Process.Start(mapsUrl);
                    break;
            }
        }
        catch (Exception ex)
        {
            MessageBox.Show($"Error handling contact interaction: {ex.Message}");
        }
    }

    private async void OnSocialLinkClicked(object sender, SocialLinkEventArgs e)
    {
        try
        {
            // Track social media interaction
            await analyticsService.TrackSocialInteractionAsync(e.Platform, e.Url);

            // Open social media link
            Process.Start(e.Url);

            // Show follow prompt for first-time interactions
            var interactionCount = await analyticsService.GetSocialInteractionCountAsync(e.Platform);
            if (interactionCount == 1)
            {
                ShowFollowPrompt(e.Platform);
            }
        }
        catch (Exception ex)
        {
            MessageBox.Show($"Error opening social link: {ex.Message}");
        }
    }

    private async void OnMetricClicked(object sender, MetricEventArgs e)
    {
        try
        {
            // Track metric interaction
            await analyticsService.TrackMetricClickAsync(e.MetricName);

            // Show detailed metric information
            var detailedMetric = await analyticsService.GetDetailedMetricAsync(e.MetricName);
            ShowMetricDetailDialog(detailedMetric);
        }
        catch (Exception ex)
        {
            MessageBox.Show($"Error loading metric details: {ex.Message}");
        }
    }

    private void OnLogoClicked(object sender, EventArgs e)
    {
        // Show company information dialog or navigate to about page
        ShowCompanyInfoDialog();
    }

    // Helper methods
    private async Task LoadImageAsync(string imageUrl)
    {
        if (string.IsNullOrEmpty(imageUrl))
            return Properties.Resources.DefaultCompanyLogo;

        try
        {
            using (var httpClient = new HttpClient())
            {
                var imageBytes = await httpClient.GetByteArrayAsync(imageUrl);
                using (var stream = new MemoryStream(imageBytes))
                {
                    return Image.FromStream(stream);
                }
            }
        }
        catch
        {
            return Properties.Resources.DefaultCompanyLogo;
        }
    }

    private SocialPlatform ParseSocialPlatform(string platform)
    {
        return platform.ToLower() switch
        {
            "linkedin" => SocialPlatform.LinkedIn,
            "twitter" => SocialPlatform.Twitter,
            "facebook" => SocialPlatform.Facebook,
            "instagram" => SocialPlatform.Instagram,
            "github" => SocialPlatform.GitHub,
            "youtube" => SocialPlatform.YouTube,
            _ => SocialPlatform.Other
        };
    }

    private bool IsEmailClientAvailable() { /* Implementation */ return true; }
    private bool IsDialerAvailable() { /* Implementation */ return false; }
    private void ShowNotification(string message) { /* Implementation */ }
    private void ShowFollowPrompt(SocialPlatform platform) { /* Implementation */ }
    private void ShowMetricDetailDialog(object detailedMetric) { /* Implementation */ }
    private void ShowCompanyInfoDialog() { /* Implementation */ }
}

Theming

BeepCompanyProfile provides comprehensive theming support with automatic adaptation to corporate branding:

Theme Application

var companyProfile = new BeepCompanyProfile
{
    Theme = "CorporateBrand",
    UseThemeColors = true
};

// Theme automatically applies:
// - HeaderBackColor: Profile header background
// - ContentBackColor: Main content area background
// - AccentColor: Metrics and highlight colors
// - TextColors: Primary and secondary text colors
// - BorderColors: Section dividers and borders
// - ButtonColors: Action button styling
