Overview

BeepDatePicker is a sophisticated date and time picker control that combines text input with a dropdown calendar interface. It provides extensive formatting options, culture-aware date handling, input masking, validation, and an integrated calendar view for date selection.

Built on the BeepControl foundation, BeepDatePicker features multiple date format styles, real-time input validation, customizable date masks, and seamless integration with the Beep theming system. The control includes a custom BeepCalendarView for visual date selection.

Basic Usage

Simple Date Picker

// Create a basic BeepDatePicker
var datePicker = new BeepDatePicker
{
    Size = new Size(200, 36),
    Location = new Point(10, 10),
    DateFormatStyle = DateFormatStyle.ShortDate
};

// Set initial date
datePicker.SelectedDateTime = DateTime.Today;

// Add to form
this.Controls.Add(datePicker);

// Handle date changes
datePicker.SelectedDateTimeChanged += (sender, e) => {
    Console.WriteLine($"Date selected: {datePicker.SelectedDateTime}");
};

Examples

Event Registration Form

public void CreateEventRegistrationForm()
{
    var panel = new BeepPanel
    {
        TitleText = "Event Registration",
        ShowTitle = true,
        Size = new Size(400, 300),
        IsRounded = true
    };

    // Event date picker
    var eventDatePicker = new BeepDatePicker
    {
        DateFormatStyle = DateFormatStyle.LongDate,
        SelectedDateTime = DateTime.Today.AddDays(7), // Default to next week
        MinDate = DateTime.Today, // Can't select past dates
        Size = new Size(350, 40),
        Location = new Point(25, 60)
    };

    // Registration deadline
    var deadlinePicker = new BeepDatePicker
    {
        DateFormatStyle = DateFormatStyle.ShortDateTime,
        SelectedDateTime = DateTime.Today.AddDays(3),
        MinDate = DateTime.Today,
        Size = new Size(250, 40),
        Location = new Point(25, 120)
    };

    // Birth date (for age verification)
    var birthDatePicker = new BeepDatePicker
    {
        DateFormatStyle = DateFormatStyle.ShortDate,
        SelectedDateTime = DateTime.Today.AddYears(-25),
        MaxDate = DateTime.Today.AddYears(-18), // Must be 18+
        Size = new Size(200, 40),
        Location = new Point(25, 180)
    };

    // Validation
    eventDatePicker.SelectedDateTimeChanged += (s, e) => {
        // Update deadline max date to be before event date
        deadlinePicker.MaxDate = eventDatePicker.SelectedDateTime.AddDays(-1);
        
        if (deadlinePicker.SelectedDateTime >= eventDatePicker.SelectedDateTime)
        {
            deadlinePicker.SelectedDateTime = eventDatePicker.SelectedDateTime.AddDays(-1);
        }
    };

    panel.Controls.AddRange(new Control[] 
    {
        new BeepLabel { Text = "Event Date:", Location = new Point(25, 40) },
        eventDatePicker,
        new BeepLabel { Text = "Registration Deadline:", Location = new Point(25, 100) },
        deadlinePicker,
        new BeepLabel { Text = "Birth Date:", Location = new Point(25, 160) },
        birthDatePicker
    });

    this.Controls.Add(panel);
}

Multi-Culture Date Pickers

public void CreateMultiCultureDatePickers()
{
    var cultures = new[]
    {
        new { Culture = new CultureInfo("en-US"), Name = "US Format" },
        new { Culture = new CultureInfo("en-GB"), Name = "UK Format" },
        new { Culture = new CultureInfo("de-DE"), Name = "German Format" },
        new { Culture = new CultureInfo("ja-JP"), Name = "Japanese Format" }
    };

    int yPosition = 20;
    foreach (var cultureInfo in cultures)
    {
        var label = new BeepLabel
        {
            Text = cultureInfo.Name,
            Location = new Point(20, yPosition),
            Size = new Size(120, 25)
        };

        var picker = new BeepDatePicker
        {
            Culture = cultureInfo.Culture,
            DateFormatStyle = DateFormatStyle.ShortDate,
            SelectedDateTime = DateTime.Today,
            Location = new Point(150, yPosition),
            Size = new Size(200, 35)
        };

        this.Controls.AddRange(new Control[] { label, picker });
        yPosition += 45;
    }
}

Appointment Scheduler

public void CreateAppointmentScheduler()
{
    // Date picker for appointment date
    var appointmentDate = new BeepDatePicker
    {
        DateFormatStyle = DateFormatStyle.LongDate,
        SelectedDateTime = DateTime.Today.AddDays(1),
        MinDate = DateTime.Today,
        MaxDate = DateTime.Today.AddMonths(3),
        Size = new Size(350, 40),
        Location = new Point(20, 60)
    };

    // Time picker for start time
    var startTime = new BeepDatePicker
    {
        DateFormatStyle = DateFormatStyle.ShortTime,
        SelectedDateTime = DateTime.Today.AddHours(9), // 9 AM default
        Size = new Size(150, 40),
        Location = new Point(20, 120)
    };

    // Time picker for end time
    var endTime = new BeepDatePicker
    {
        DateFormatStyle = DateFormatStyle.ShortTime,
        SelectedDateTime = DateTime.Today.AddHours(10), // 10 AM default
        Size = new Size(150, 40),
        Location = new Point(200, 120)
    };

    // Validation for business hours
    startTime.SelectedDateTimeChanged += (s, e) => {
        var time = startTime.SelectedDateTime.TimeOfDay;
        if (time < TimeSpan.FromHours(8) || time > TimeSpan.FromHours(17))
        {
            MessageBox.Show("Please select a time between 8 AM and 5 PM");
            startTime.SelectedDateTime = DateTime.Today.AddHours(9);
        }
        
        // Update end time to be 1 hour later
        endTime.SelectedDateTime = startTime.SelectedDateTime.AddHours(1);
    };

    // Prevent weekend appointments
    appointmentDate.SelectedDateTimeChanged += (s, e) => {
        var date = appointmentDate.SelectedDateTime;
        if (date.DayOfWeek == DayOfWeek.Saturday || date.DayOfWeek == DayOfWeek.Sunday)
        {
            MessageBox.Show("Appointments are not available on weekends");
            // Move to next Monday
            while (date.DayOfWeek == DayOfWeek.Saturday || date.DayOfWeek == DayOfWeek.Sunday)
            {
                date = date.AddDays(1);
            }
            appointmentDate.SelectedDateTime = date;
        }
    };

    this.Controls.AddRange(new Control[] 
    {
        new BeepLabel { Text = "Appointment Date:", Location = new Point(20, 40) },
        appointmentDate,
        new BeepLabel { Text = "Start Time:", Location = new Point(20, 100) },
        startTime,
        new BeepLabel { Text = "End Time:", Location = new Point(200, 100) },
        endTime
    });
}

};

Localization

BeepDatePicker provides comprehensive localization support through the .NET CultureInfo system.

Culture-Specific Formatting

// Create date pickers for different cultures
var cultures = new[]
{
    new CultureInfo("en-US"), // MM/dd/yyyy
    new CultureInfo("en-GB"), // dd/MM/yyyy
    new CultureInfo("de-DE"), // dd.MM.yyyy
    new CultureInfo("fr-FR"), // dd/MM/yyyy
    new CultureInfo("ja-JP"), // yyyy/MM/dd
    new CultureInfo("ar-SA")  // dd/MM/yyyy (RTL)
};

foreach (var culture in cultures)
{
    var picker = new BeepDatePicker
    {
        Culture = culture,
        DateFormatStyle = DateFormatStyle.ShortDate,
        SelectedDateTime = DateTime.Today,
        Size = new Size(200, 40)
    };
    
    // The placeholder text automatically adjusts to the culture
    Console.WriteLine($"{culture.Name}: {picker.GetPlaceholderText()}");
}

Dynamic Culture Switching

var datePicker = new BeepDatePicker
{
    DateFormatStyle = DateFormatStyle.ShortDate,
    SelectedDateTime = DateTime.Today
};

// Culture selector
var cultureCombo = new BeepComboBox
{
    PlaceholderText = "Select culture..."
};

var supportedCultures = new[]
{
    new { Name = "English (US)", Code = "en-US" },
    new { Name = "English (UK)", Code = "en-GB" },
    new { Name = "German", Code = "de-DE" },
    new { Name = "French", Code = "fr-FR" },
    new { Name = "Japanese", Code = "ja-JP" }
};

foreach (var culture in supportedCultures)
{
    cultureCombo.ListItems.Add(new SimpleItem 
    { 
        Text = culture.Name, 
        Value = culture.Code 
    });
}

cultureCombo.SelectedItemChanged += (s, e) => {
    if (e.SelectedItem?.Value is string cultureCode)
    {
        datePicker.Culture = new CultureInfo(cultureCode);
        // Date format and placeholder automatically update
    }
};

Best Practices

üé® User Experience

  • Choose appropriate formats: Use date-only for birth dates, date-time for appointments
  • Provide clear labels: Always label date pickers with their purpose
  • Set sensible defaults: Pre-populate with logical default dates
  • Validate ranges: Implement min/max date validation for business rules
  • Culture awareness: Use appropriate culture settings for your user base

üìÖ Date Handling

  • Handle edge cases: Account for leap years, month boundaries, and time zones
  • Consistent formatting: Use the same date format throughout your application
  • Validation feedback: Provide clear error messages for invalid dates
  • Future vs. past: Restrict date ranges based on business logic

‚ôø Accessibility

  • Keyboard navigation: Ensure full keyboard accessibility
  • Screen readers: Use meaningful labels and descriptions
  • High contrast: Ensure sufficient contrast for date display
  • Touch targets: Make dropdown buttons large enough for touch input

‚ö° Performance

  • Event handling: Avoid excessive validation on every keystroke
  • Calendar popup: Cache calendar instances for better performance
  • Culture objects: Reuse CultureInfo instances when possible
‚ö†Ô∏è Culture Warning

When working with different cultures, be aware of date parsing ambiguities. Always specify the culture explicitly for consistency.

‚úÖ Success Pattern

Always validate both the format and business logic of selected dates, providing clear feedback to users about any issues.
