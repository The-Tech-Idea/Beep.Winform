Overview

BeepGrid is a sophisticated data grid control that extends the functionality of standard DataGridView with advanced features for enterprise applications. It provides custom column types, integrated filtering, sorting, totals calculation, printing capabilities, and comprehensive data management tools.

Built on the BeepControl foundation, BeepGrid features a modular panel system with customizable header, filter, totals, and navigation panels. It supports dynamic column creation from EntityStructure, layout persistence, and seamless integration with data sources through the Beep framework.

Basic Usage

Simple Data Grid

// Create a basic BeepGrid
var grid = new BeepGrid
{
    Size = new Size(800, 400),
    Location = new Point(10, 10),
    ShowHeaderPanel = true,
    ShowFilterPanel = false,
    ShowTotalsPanel = false,
    ShowNavigatorPanel = true
};

// Set data source
var dataTable = GetSampleData();
grid.DataSource = dataTable;

// Add to form
this.Controls.Add(grid);

Grid with Entity Structure

// Create grid with entity-based columns
var grid = new BeepGrid();
grid.SetConfig(dmeEditor, logger, util, null, null, null);

// Define entity structure
var entityStructure = new EntityStructure
{
    EntityName = "Employees",
    Fields = new List
    {
        new EntityField { fieldname = "ID", fieldtype = "System.Int32" },
        new EntityField { fieldname = "Name", fieldtype = "System.String" },
        new EntityField { fieldname = "Salary", fieldtype = "System.Decimal" },
        new EntityField { fieldname = "HireDate", fieldtype = "System.DateTime" },
        new EntityField { fieldname = "IsActive", fieldtype = "System.Boolean" }
    }
};

// Reset grid with data and entity
var employees = GetEmployeeData();
grid.ResetData(employees, entityStructure);

Grid with Panel Configuration

var grid = new BeepGrid
{
    Size = new Size(1000, 600),
    
    // Panel visibility
    ShowHeaderPanel = true,      // Show title and action buttons
    ShowColumnHeaderPanel = true, // Show custom column headers
    ShowFilterPanel = true,       // Show filter row
    ShowTotalsPanel = true,       // Show totals row
    ShowNavigatorPanel = true,    // Show navigation controls
    
    // Grid properties
    AllowUserToAddRows = false,
    AllowUserToDeleteRows = true,
    ReadOnly = false
};

// Configure title
grid.Title.Text = "Employee Management Grid";

// Handle events
grid.CellDoubleClick += (s, e) => {
    if (e.RowIndex >= 0)
    {
        var employee = GetEmployeeFromRow(e.RowIndex);
        EditEmployee(employee);
    }
};

Event Handling Examples

var grid = new BeepGrid();

// Handle cell interactions
grid.CellDoubleClick += (s, e) => {
    if (e.RowIndex >= 0 && e.ColumnIndex >= 0)
    {
        var value = grid[e.ColumnIndex, e.RowIndex].Value;
        Console.WriteLine($"Double-clicked: {value}");
        
        // Open edit dialog
        EditRecord(e.RowIndex);
    }
};

// Handle data operations
grid.SaveCalled += (s, e) => {
    try
    {
        SaveChangesToDatabase(e.DataSource);
        MessageBox.Show("Changes saved successfully");
    }
    catch (Exception ex)
    {
        MessageBox.Show($"Save failed: {ex.Message}");
    }
};

// Handle formatting
grid.CellFormatting += (s, e) => {
    if (e.ColumnIndex == salaryColumnIndex && e.Value != null)
    {
        if (decimal.TryParse(e.Value.ToString(), out decimal salary))
        {
            e.CellStyle.BackColor = salary > 50000 ? Color.LightGreen : Color.LightPink;
            e.FormattingApplied = true;
        }
    }
};

{

Dynamic Configuration Grid

public void CreateConfigurableGrid()
{
    var configurableGrid = new BeepGrid
    {
        GridId = "ConfigurableDataGrid", // For layout persistence
        Size = new Size(900, 500)
    };

    // Column configurations for advanced scenarios
    var columnConfigs = new List
    {
        new BeepColumnConfig
        {
            ColumnName = "ID",
            ColumnCaption = "Record ID",
            Width = 80,
            IsFilteOn = false,
            IsTotalOn = false
        },
        new BeepColumnConfig
        {
            ColumnName = "Amount",
            ColumnCaption = "Transaction Amount",
            Width = 120,
            IsFilteOn = true,
            IsTotalOn = true,
            HasTotal = true
        }
    };

    configurableGrid.ColumnsConfigurations = columnConfigs;

    // Enable all panels for full functionality
    configurableGrid.ShowHeaderPanel = true;
    configurableGrid.ShowFilterPanel = true;
    configurableGrid.ShowTotalsPanel = true;
    configurableGrid.ShowNavigatorPanel = true;

    // Load and save layout automatically
    configurableGrid.LoadGridLayout();
    
    // Save layout when form closes
    this.FormClosing += (s, e) => {
        configurableGrid.SaveGridLayout();
    };
}

Filtering & Sorting

BeepGrid provides powerful filtering and sorting capabilities:

Enable Filtering

// Show filter panel
grid.ShowFilterPanel = true;

// Filters are automatically available for each column
// Users can type in filter boxes to filter data

// Programmatic filtering example
private void ApplyCustomFilter()
{
    // Access the binding source for complex filtering
    var bindingSource = grid.BindingNavigator.BindingSource;
    
    // Apply complex filter
    bindingSource.Filter = "Salary > 50000 AND Department = 'Engineering'";
    
    // Remove filter
    bindingSource.RemoveFilter();
}

Column Sorting

// Sorting is automatically enabled on column headers
// Click to sort, click again to reverse, third click to remove sort

// Programmatic sorting
grid.Sort(grid.Columns["Salary"], ListSortDirection.Descending);

// Handle sort events
grid.GridView.Sorted += (s, e) => {
    Console.WriteLine("Grid sorted");
    UpdateTotalCalculations();
};

Totals Calculation

// Enable totals panel
grid.ShowTotalsPanel = true;

// Totals are automatically calculated for numeric columns
// Custom total calculation can be implemented through column configs

// Access calculated totals
foreach (var config in grid.ColumnsConfigurations)
{
    if (config.HasTotal && config.IsTotalOn)
    {
        Console.WriteLine($"{config.ColumnName}: {config.Total}");
    }
}

Best Practices

‚ö° Performance

  • Virtual Mode: Use virtual mode for large datasets (>10,000 rows)
  • Column Types: Use appropriate column types for better performance and UX
  • Data Binding: Bind to proper data sources (DataTable, BindingList, etc.)
  • Layout Persistence: Use GridId for automatic layout saving/loading
  • Suspend Layout: Suspend layout during bulk column operations

üìä Data Management

  • Entity Structure: Define proper EntityStructure for automatic column generation
  • Data Validation: Implement cell validation for data integrity
  • Error Handling: Handle DataError events for robust applications
  • Transaction Support: Use transactions for data modifications

üé® User Experience

  • Panel Configuration: Show/hide panels based on user needs
  • Column Sizing: Set appropriate default column widths
  • Filtering: Enable filtering for searchable data sets
  • Sorting: Ensure sortable columns are properly configured
  • Visual Feedback: Use formatting and custom columns for better data visualization

‚ôø Accessibility

  • Keyboard Navigation: Ensure full keyboard accessibility
  • Screen Readers: Use meaningful column headers and descriptions
  • High Contrast: Test with high contrast themes
  • Focus Indicators: Provide clear focus indication
‚ö†Ô∏è Performance Warning

Be cautious with totals panel on very large datasets as it can impact performance. Consider implementing lazy loading or virtual totals for large data sets.
