BeepNumericUpDown

Composite numeric input control with text box, increment/decrement buttons, and validation

Overview

BeepNumericUpDown is a composite control that combines a BeepTextBox for numeric input with increment and decrement BeepButton controls. It provides a user-friendly interface for numeric data entry with built-in validation, range checking, and customizable increment values.

This control automatically manages the layout and interaction between its child components, providing a seamless experience for numeric input while maintaining full theme compatibility and accessibility features.

Basic Usage

Simple Numeric Input

// Create a basic numeric up/down control
var numericControl = new BeepNumericUpDown
{
    Value = 10,
    MinimumValue = 0,
    MaximumValue = 100,
    IncrementValue = 1,
    Size = new Size(120, 30),
    Location = new Point(10, 10)
};

// Add to form
this.Controls.Add(numericControl);

// Handle value changes
numericControl.ValueChanged += (sender, e) => {
    Console.WriteLine($"Value changed to: {numericControl.Value}");
};

Event Handling Examples

var numericControl = new BeepNumericUpDown
{
    Value = 50,
    MinimumValue = 0,
    MaximumValue = 100
};

// Handle value changes
numericControl.ValueChanged += (sender, e) => {
    Console.WriteLine($"New value: {numericControl.Value}");
    UpdateDependentControls(numericControl.Value);
};

// Validate values before accepting changes
numericControl.ValueValidating += (sender, e) => {
    decimal newValue = numericControl.Value;
    
    if (newValue % 5 != 0) // Only allow multiples of 5
    {
        e.Cancel = true;
        MessageBox.Show("Value must be a multiple of 5", "Invalid Value");
    }
};

// Handle range limits

    MessageBox.Show("Minimum value reached!", "Range Limit");
    ShowMinimumWarning();
};

numericControl.MaximumReached += (sender, e) => {
    MessageBox.Show("Maximum value reached!", "Range Limit");

Settings Configuration Panel

public class SettingsPanel : UserControl
{
    private BeepNumericUpDown timeoutControl;
    private BeepNumericUpDown maxConnectionsControl;
    private BeepNumericUpDown bufferSizeControl;
    private BeepNumericUpDown refreshRateControl;

    public SettingsPanel()
    {
        InitializeSettingsControls();
    }

    private void InitializeSettingsControls()
    {
        // Timeout setting (seconds)
        timeoutControl = new BeepNumericUpDown
        {
            Value = 30,
            MinimumValue = 5,
            MaximumValue = 300,
            IncrementValue = 5,
            Suffix = " sec",
            Location = new Point(150, 20),
            Size = new Size(120, 30)
        };

        // Max connections
        maxConnectionsControl = new BeepNumericUpDown
        {
            Value = 100,
            MinimumValue = 10,
            MaximumValue = 1000,
            IncrementValue = 10,
            Location = new Point(150, 60),
            Size = new Size(100, 30)
        };

        // Buffer size (MB)
        bufferSizeControl = new BeepNumericUpDown
        {
            Value = 64,
            MinimumValue = 16,
            MaximumValue = 512,
            IncrementValue = 16,
            Suffix = " MB",
            Location = new Point(150, 100),
            Size = new Size(120, 30)
        };

        // Refresh rate (Hz)
        refreshRateControl = new BeepNumericUpDown
        {
            Value = 60,
            MinimumValue = 30,
            MaximumValue = 144,
            IncrementValue = 15,
            Suffix = " Hz",
            Location = new Point(150, 140),
            Size = new Size(120, 30)
        };

        // Validation and interdependency logic
        SetupValidationRules();
        
        this.Controls.AddRange(new Control[]
        {
            new Label { Text = "Connection Timeout:", Location = new Point(20, 25) },
            timeoutControl,
            new Label { Text = "Max Connections:", Location = new Point(20, 65) },
            maxConnectionsControl,
            new Label { Text = "Buffer Size:", Location = new Point(20, 105) },
            bufferSizeControl,
            new Label { Text = "Refresh Rate:", Location = new Point(20, 145) },
            refreshRateControl
        });
    }

    private void SetupValidationRules()
    {
        // Interdependent validation
        bufferSizeControl.ValueChanged += (s, e) => {
            // Adjust max connections based on buffer size
            decimal bufferSize = bufferSizeControl.Value;
            decimal maxConnections = Math.Min(1000, bufferSize * 10);
            
            if (maxConnectionsControl.Value > maxConnections)
            {
                maxConnectionsControl.Value = maxConnections;
                MessageBox.Show($"Max connections adjusted to {maxConnections} based on buffer size.",
                    "Setting Adjusted", MessageBoxButtons.OK, MessageBoxIcon.Information);
            }
        };

        // Performance warning for high refresh rates
        refreshRateControl.ValueChanged += (s, e) => {
            if (refreshRateControl.Value > 120)
            {
                MessageBox.Show("High refresh rates may impact system performance.",
                    "Performance Warning", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            }
        };
    }

    public Dictionary GetSettings()
    {
        return new Dictionary
        {
            ["Timeout"] = (int)timeoutControl.Value,
            ["MaxConnections"] = (int)maxConnectionsControl.Value,
            ["BufferSize"] = (int)bufferSizeControl.Value,
            ["RefreshRate"] = (int)refreshRateControl.Value
        };
    }

    public void LoadSettings(Dictionary settings)
    {
        if (settings.TryGetValue("Timeout", out var timeout))
            timeoutControl.Value = Convert.ToDecimal(timeout);
            
        if (settings.TryGetValue("MaxConnections", out var maxConn))
            maxConnectionsControl.Value = Convert.ToDecimal(maxConn);
            
        if (settings.TryGetValue("BufferSize", out var bufferSize))
            bufferSizeControl.Value = Convert.ToDecimal(bufferSize);
            
        if (settings.TryGetValue("RefreshRate", out var refreshRate))
            refreshRateControl.Value = Convert.ToDecimal(refreshRate);
    }
}

Scientific Calculator Input

public class ScientificCalculatorInput : UserControl
{
    private BeepNumericUpDown mantissaControl;
    private BeepNumericUpDown exponentControl;
    private BeepNumericUpDown precisionControl;
    private Label resultLabel;

    public ScientificCalculatorInput()
    {
        InitializeScientificControls();
    }

    private void InitializeScientificControls()
    {
        // Mantissa (coefficient)
        mantissaControl = new BeepNumericUpDown
        {
            Value = 1.000000m,
            MinimumValue = -9.999999m,
            MaximumValue = 9.999999m,
            IncrementValue = 0.000001m,
            DecimalPlaces = 6,
            Location = new Point(100, 20),
            Size = new Size(150, 30)
        };

        // Exponent
        exponentControl = new BeepNumericUpDown
        {
            Value = 0,
            MinimumValue = -308,
            MaximumValue = 308,
            IncrementValue = 1,
            Prefix = "E",
            Location = new Point(100, 60),
            Size = new Size(120, 30)
        };

        // Precision (decimal places)
        precisionControl = new BeepNumericUpDown
        {
            Value = 6,
            MinimumValue = 0,
            MaximumValue = 15,
            IncrementValue = 1,
            Location = new Point(100, 100),
            Size = new Size(100, 30)
        };

        resultLabel = new Label
        {
            Text = "Result: 1.000000",
            Location = new Point(20, 140),
            Size = new Size(300, 25),
            Font = new Font("Consolas", 10, FontStyle.Bold)
        };

        // Calculate result when any value changes
        mantissaControl.ValueChanged += CalculateResult;
        exponentControl.ValueChanged += CalculateResult;
        precisionControl.ValueChanged += UpdatePrecision;

        this.Controls.AddRange(new Control[]
        {
            new Label { Text = "Mantissa:", Location = new Point(20, 25) },
            mantissaControl,
            new Label { Text = "Exponent:", Location = new Point(20, 65) },
            exponentControl,
            new Label { Text = "Precision:", Location = new Point(20, 105) },
            precisionControl,
            resultLabel
        });
    }

    private void CalculateResult(object sender, EventArgs e)
    {
        try
        {
            double mantissa = (double)mantissaControl.Value;
            int exponent = (int)exponentControl.Value;
            int precision = (int)precisionControl.Value;
            
            double result = mantissa * Math.Pow(10, exponent);
            string format = $"F{precision}";
            
            resultLabel.Text = $"Result: {result.ToString(format)}";
            
            // Handle overflow/underflow
            if (double.IsInfinity(result))
            {
                resultLabel.Text = "Result: Infinity";
                resultLabel.ForeColor = Color.Red;
            }
            else if (result == 0 && mantissa != 0)
            {
                resultLabel.Text = "Result: Underflow";
                resultLabel.ForeColor = Color.Orange;
            }
            else
            {
                resultLabel.ForeColor = Color.Black;
            }
        }
        catch (OverflowException)
        {
            resultLabel.Text = "Result: Overflow";
            resultLabel.ForeColor = Color.Red;
        }
    }

    private void UpdatePrecision(object sender, EventArgs e)
    {
        int newPrecision = (int)precisionControl.Value;
        
        // Update mantissa decimal places to match precision
        mantissaControl.DecimalPlaces = newPrecision;
        
        // Recalculate result with new precision
        CalculateResult(sender, e);
    }

    public double GetScientificValue()
    {
        double mantissa = (double)mantissaControl.Value;
        int exponent = (int)exponentControl.Value;
        return mantissa * Math.Pow(10, exponent);
    }

    public void SetScientificValue(double value)
    {
        if (value == 0)
        {
            mantissaControl.Value = 0;
            exponentControl.Value = 0;
            return;
        }

        int exponent = (int)Math.Floor(Math.Log10(Math.Abs(value)));
        double mantissa = value / Math.Pow(10, exponent);
        
        // Ensure mantissa is within valid range
        while (Math.Abs(mantissa) >= 10)
        {
            mantissa /= 10;
            exponent++;
        }

        mantissaControl.Value = (decimal)mantissa;
        exponentControl.Value = exponent;
    }
}

  • Use reasonable increments: Choose increment values that make sense for your data type
  • Handle validation events: Implement custom validation logic when values change
  • Provide feedback: Show users when values are invalid or out of range
  • Consider data types: Use appropriate decimal places for your numeric precision needs

Validation Best Practices

public class ValidatedNumericInput : BeepNumericUpDown
{
    public event EventHandler ValidationFailed;
    
    protected override void OnValueChanged(EventArgs e)
    {
        // Validate business rules before calling base
        if (!ValidateBusinessRules(Value))
        {
            ValidationFailed?.Invoke(this, new ValidationEventArgs 
            { 
                InvalidValue = Value,
                ErrorMessage = "Value violates business rules" 
            });
            return;
        }
        
        base.OnValueChanged(e);
    }
    
    private bool ValidateBusinessRules(decimal value)
    {
        // Example: Only allow even numbers
        return value % 2 == 0;
    }
    
    public void SetValueWithValidation(decimal newValue, bool showErrors = true)
    {
        if (newValue < MinimumValue || newValue > MaximumValue)
        {
            if (showErrors)
            {
                MessageBox.Show($"Value must be between {MinimumValue} and {MaximumValue}",
                    "Range Validation", MessageBoxButtons.OK, MessageBoxIcon.Warning);
            }
            return;
        }
        
        Value = newValue;
    }
}
