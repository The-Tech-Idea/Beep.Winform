Fully custom tab control with themed headers, closeable tabs, drag-and-drop reordering, and SVG close buttons

Overview

BeepTabs is a fully custom tab control that extends the standard TabControl with advanced features including themed headers, closeable tabs, drag-and-drop tab reordering, and flexible header positioning. Unlike traditional tab controls, BeepTabs provides complete custom rendering with owner-drawn headers and smooth anti-aliased graphics.

Built as an enhanced TabControl, BeepTabs features SVG close buttons, rounded corners, custom color schemes, and supports header positioning on all four sides (Top, Bottom, Left, Right). It includes comprehensive drag-and-drop functionality for tab reordering and integrates seamlessly with the Beep theming system.

Basic Usage

Simple Tab Control

// Create a basic BeepTabs control
var tabControl = new BeepTabs
{
    Size = new Size(500, 400),
    Location = new Point(10, 10),
    HeaderPosition = TabHeaderPosition.Top,
    HeaderHeight = 30
};

// Create and add tab pages
var tabPage1 = new TabPage("General")
{
    BackColor = Color.White
};
var tabPage2 = new TabPage("Advanced")
{
    BackColor = Color.White
};
var tabPage3 = new TabPage("Settings")
{
    BackColor = Color.White
};

// Add content to tabs
var label1 = new Label { Text = "General Settings", Dock = DockStyle.Fill };
var label2 = new Label { Text = "Advanced Options", Dock = DockStyle.Fill };
var label3 = new Label { Text = "Configuration", Dock = DockStyle.Fill };

tabPage1.Controls.Add(label1);
tabPage2.Controls.Add(label2);
tabPage3.Controls.Add(label3);

// Add tabs to control
tabControl.TabPages.Add(tabPage1);
tabControl.TabPages.Add(tabPage2);
tabControl.TabPages.Add(tabPage3);

// Add to form
this.Controls.Add(tabControl);

// Dynamic header position change
private void ChangeHeaderPosition()
{
    switch (currentPosition)
    {
        case TabHeaderPosition.Top:
            tabControl.HeaderPosition = TabHeaderPosition.Right;
            tabControl.HeaderHeight = 150;
            break;
        case TabHeaderPosition.Right:
            tabControl.HeaderPosition = TabHeaderPosition.Bottom;
            tabControl.HeaderHeight = 30;
            break;
        case TabHeaderPosition.Bottom:
            tabControl.HeaderPosition = TabHeaderPosition.Left;
            tabControl.HeaderHeight = 150;
            break;
        case TabHeaderPosition.Left:
            tabControl.HeaderPosition = TabHeaderPosition.Top;
            tabControl.HeaderHeight = 30;
            break;
    }
}

// Handle mouse events from parent container
protected override void OnMouseClick(MouseEventArgs e)
{
    base.OnMouseClick(e);
    
    // Forward click to tab control if within bounds
    if (tabControl.Bounds.Contains(e.Location))
    {
        Point relativeLocation = new Point(
            e.Location.X - tabControl.Left,
            e.Location.Y - tabControl.Top
        );
        tabControl.ReceiveMouseClick(relativeLocation);
    }
}

// Advanced tab management
private void CleanupTabResources(string tabText)
{
    // Dispose of any resources associated with the tab
    if (tabResources.ContainsKey(tabText))
    {
        var resources = tabResources[tabText];
        resources.Dispose();
        tabResources.Remove(tabText);
    }
}

private void LoadTabContent(TabPage tabPage)
{
    // Lazy load content when tab is selected
    if (tabPage.Controls.Count == 0)
    {
        // Create and add content
        var content = CreateContentForTab(tabPage.Text);
        if (content != null)
        {
            tabPage.Controls.Add(content);
        }
    }
}

Examples

Document Editor with Closeable Tabs

public class DocumentEditor : Form
{
    private BeepTabs documentTabs;
    private Dictionary openDocuments;

    public DocumentEditor()
    {
        InitializeDocumentTabs();
        openDocuments = new Dictionary();
    }

    private void InitializeDocumentTabs()
    {
        documentTabs = new BeepTabs
        {
            Dock = DockStyle.Fill,
            HeaderPosition = TabHeaderPosition.Top,
            HeaderHeight = 32,
            Theme = "DefaultTheme"
        };

        // Handle tab removal
        documentTabs.TabRemoved += DocumentTabs_TabRemoved;
        documentTabs.SelectedIndexChanged += DocumentTabs_SelectedIndexChanged;

        this.Controls.Add(documentTabs);
    }

    public void OpenDocument(string filePath)
    {
        string fileName = Path.GetFileName(filePath);
        
        // Check if document is already open
        foreach (TabPage existingTab in documentTabs.TabPages)
        {
            if (existingTab.Text == fileName)
            {
                documentTabs.SelectedTab = existingTab;
                return;
            }
        }

        // Create new tab for document
        var tabPage = new TabPage(fileName);
        var textEditor = new RichTextBox
        {
            Dock = DockStyle.Fill,
            Font = new Font("Consolas", 11f),
            Text = File.ReadAllText(filePath)
        };

        // Track document state
        var docData = new DocumentData
        {
            FilePath = filePath,
            Editor = textEditor,
            IsModified = false
        };

        textEditor.TextChanged += (s, e) => {
            docData.IsModified = true;
            UpdateTabTitle(tabPage, fileName);
        };

        tabPage.Controls.Add(textEditor);
        documentTabs.TabPages.Add(tabPage);
        documentTabs.SelectedTab = tabPage;

        openDocuments[fileName] = docData;
    }

    private void DocumentTabs_TabRemoved(object sender, TabRemovedEventArgs e)
    {
        string fileName = e.TabText.TrimEnd('*'); // Remove modified indicator

        if (openDocuments.ContainsKey(fileName))
        {
            var docData = openDocuments[fileName];
            
            // Check for unsaved changes
            if (docData.IsModified)
            {
                var result = MessageBox.Show(
                    $"Save changes to {fileName}?",
                    "Unsaved Changes",
                    MessageBoxButtons.YesNoCancel,
                    MessageBoxIcon.Question);

                switch (result)
                {
                    case DialogResult.Yes:
                        SaveDocument(docData);
                        break;
                    case DialogResult.Cancel:
                        // Re-add the tab (user cancelled)
                        var tabPage = new TabPage(e.TabText);
                        tabPage.Controls.Add(docData.Editor);
                        documentTabs.TabPages.Add(tabPage);
                        return;
                }
            }

            openDocuments.Remove(fileName);
        }

        // Show welcome screen if no documents open
        if (documentTabs.TabCount == 0)
        {
            ShowWelcomeTab();
        }
    }

    private class DocumentData
    {
        public string FilePath { get; set; }
        public RichTextBox Editor { get; set; }
        public bool IsModified { get; set; }
    }
}

Theming

BeepTabs provides comprehensive theming support for all visual elements:

Apply Theme

var tabControl = new BeepTabs
{
    Theme = "DarkTheme"
};

// Theme automatically applies:
// - TabBackColor (inactive tab background)
// - TabForeColor (inactive tab text)
// - TabSelectedBackColor (active tab background)
// - TabSelectedForeColor (active tab text)

// Apply theme manually if needed
tabControl.ApplyTheme();

Custom Theme Integration

// Apply theme to nested controls
tabControl.ApplyTheme();

// Theme is automatically applied to:
// - All TabPage backgrounds
// - Child controls implementing IBeepUIComponent
// - Addin controls and their children

// Custom theme application for specific scenarios
foreach (TabPage tabPage in tabControl.TabPages)
{
    tabPage.BackColor = currentTheme.TabBackColor;
    tabPage.ForeColor = currentTheme.TabForeColor;
    
    foreach (Control control in tabPage.Controls)
    {
        if (control is IBeepUIComponent beepControl)
        {
            beepControl.Theme = tabControl.Theme;
        }
    }
}

Best Practices

üé® User Experience

  • Tab Limits: Consider limiting the number of tabs to prevent overcrowding
  • Close Confirmation: Implement confirmation for closing modified content
  • Tab Ordering: Provide logical tab ordering and allow user customization
  • Visual Feedback: Use modified indicators (*) for unsaved content
  • Keyboard Navigation: Support Ctrl+Tab for tab switching

‚ö° Performance

  • Lazy Loading: Load tab content only when selected
  • Resource Management: Dispose of tab resources when tabs are closed
  • Layout Suspension: Use SuspendFormLayout during bulk operations
  • Memory Optimization: Clear unused tab references

üé® Visual Design

  • Header Position: Choose appropriate header position for your layout
  • Header Sizing: Set appropriate HeaderHeight for readability
  • Theme Consistency: Apply consistent theming across all tabs
  • Icon Usage: Consider adding icons to tab headers for better identification

‚ôø Accessibility

  • Keyboard Support: Ensure full keyboard navigation
  • Screen Readers: Use meaningful tab text and descriptions
  • Focus Indicators: Provide clear focus indication
  • High Contrast: Test with high contrast themes
‚ö†Ô∏è Performance Warning

Be careful with tab limits and resource management. Too many tabs can impact performance and user experience.

‚úÖ Success Pattern

Use BeepTabs for document interfaces, settings dialogs, and any scenario requiring organized content sections with closeable tabs.
