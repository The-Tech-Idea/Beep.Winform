Overview

BeepWizard provides a complete wizard framework designed to guide users through multi-step processes with ease and elegance. It combines powerful step management, smooth slide animations, multiple layout options, and comprehensive validation support to create professional wizard experiences for applications.

Built on the BeepControl foundation, BeepWizard features intelligent step navigation, three distinct view layouts, logo and branding support, slide transition animations, and extensive theming integration. Perfect for setup wizards, onboarding flows, data collection forms, and any guided user process.

Basic Usage

Simple Setup Wizard

// Create wizard with breadcrumb layout
var setupWizard = new BeepWizard
{
    Size = new Size(700, 500),
    Location = new Point(50, 50),
    ViewType = WizardViewType.TopBreadcrumb,
    Title = "Application Setup",
    Description = "Configure your application settings",
    LogoPath = "Resources/app-logo.svg"
};

// Create wizard steps (implement IWizardNode)
var welcomeStep = new WelcomeWizardNode();
var userInfoStep = new UserInfoWizardNode();
var preferencesStep = new PreferencesWizardNode();
var reviewStep = new ReviewWizardNode();
var completeStep = new CompleteWizardNode();

// Add steps to wizard
setupWizard.AddSteps(new[] {
    welcomeStep,
    userInfoStep,
    preferencesStep,
    reviewStep,
    completeStep
});

// Handle wizard events
setupWizard.NodeChanging += OnWizardNodeChanging;
setupWizard.NodeChanged += OnWizardNodeChanged;
setupWizard.FinishRequested += OnWizardFinishRequested;
setupWizard.CancelRequested += OnWizardCancelRequested;

this.Controls.Add(setupWizard);

Creating Wizard Nodes

public class UserInfoWizardNode : IWizardNode
{
    public string Name => "User Information";
    public Control Page { get; private set; }
    public bool CanMoveNext => ValidateUserInfo();
    public bool CanMovePrevious => true;
    public bool CanFinish => false;
    public bool CanCancel => true;

    private BeepTextBox txtFirstName;
    private BeepTextBox txtLastName;
    private BeepTextBox txtEmail;

    public UserInfoWizardNode()
    {
        InitializePage();
    }

    private void InitializePage()
    {
        Page = new Panel { Dock = DockStyle.Fill };
        
        var titleLabel = new BeepLabel
        {
            Text = "Enter Your Information",
            Font = new Font("Segoe UI", 16, FontStyle.Bold),
            Location = new Point(20, 20),
            Size = new Size(400, 30)
        };

        txtFirstName = new BeepTextBox
        {
            PlaceholderText = "First Name",
            Location = new Point(20, 70),
            Size = new Size(200, 30)
        };

        txtLastName = new BeepTextBox
        {
            PlaceholderText = "Last Name",
            Location = new Point(240, 70),
            Size = new Size(200, 30)
        };

        txtEmail = new BeepTextBox
        {
            PlaceholderText = "Email Address",
            Location = new Point(20, 120),
            Size = new Size(420, 30)
        };

        // Add validation events
        txtFirstName.TextChanged += (s, e) => OnValidationChanged();
        txtLastName.TextChanged += (s, e) => OnValidationChanged();
        txtEmail.TextChanged += (s, e) => OnValidationChanged();

        Page.Controls.AddRange(new Control[]
        {
            titleLabel, txtFirstName, txtLastName, txtEmail
        });
    }

    private bool ValidateUserInfo()
    {
        return !string.IsNullOrWhiteSpace(txtFirstName.Text) &&
               !string.IsNullOrWhiteSpace(txtLastName.Text) &&
               !string.IsNullOrWhiteSpace(txtEmail.Text) &&
               IsValidEmail(txtEmail.Text);
    }

    private bool IsValidEmail(string email)
    {
        try
        {
            var addr = new System.Net.Mail.MailAddress(email);
            return addr.Address == email;
        }
        catch
        {
            return false;
        }
    }

    private void OnValidationChanged()
    {
        // Notify wizard that validation state may have changed
        ValidationChanged?.Invoke(this, EventArgs.Empty);
    }

    public event EventHandler ValidationChanged;
}

Event Handlers

private void OnWizardNodeChanging(object sender, NodeChangeEventArgs e)
{
    // Validate before allowing navigation
    if (!e.CurrentNode.CanMoveNext && e.ToNode != null)
    {
        var result = MessageBox.Show(
            "Current step is not complete. Continue anyway?",
            "Validation Warning",
            MessageBoxButtons.YesNo,
            MessageBoxIcon.Warning);
            
        if (result == DialogResult.No)
        {
            e.Cancel = true;
        }
    }
}

private void OnWizardNodeChanged(object sender, NodeChangeEventArgs e)
{
    // Update UI or perform actions after step change
    Console.WriteLine($"Moved from {e.CurrentNode.Name} to {e.ToNode.Name}");
    
    // Save step data or update progress
    SaveStepData(e.CurrentNode);
}

private void OnWizardFinishRequested(object sender, EventArgs e)
{
    // Complete the wizard process
    if (ValidateAllSteps())
    {
        CompleteWizardProcess();
        this.Close();
    }
    else
    {
        MessageBox.Show("Please complete all required steps.", "Validation Error");
    }
}

private void OnWizardCancelRequested(object sender, EventArgs e)
{
    var result = MessageBox.Show(
        "Are you sure you want to cancel the setup? All progress will be lost.",
        "Cancel Setup",
        MessageBoxButtons.YesNo,
        MessageBoxIcon.Question);
        
    if (result == DialogResult.Yes)
    {
        this.Close();
    }
}

Examples

Complete Application Setup Wizard

public class ApplicationSetupWizard : Form
{
    private BeepWizard wizard;
    private WizardData wizardData;

    public ApplicationSetupWizard()
    {
        InitializeWizard();
        SetupWizardSteps();
    }

    private void InitializeWizard()
    {
        this.Size = new Size(800, 600);
        this.Text = "Application Setup Wizard";
        this.StartPosition = FormStartPosition.CenterScreen;
        this.FormBorderStyle = FormBorderStyle.FixedDialog;
        this.MaximizeBox = false;

        wizardData = new WizardData();

        wizard = new BeepWizard
        {
            Dock = DockStyle.Fill,
            ViewType = WizardViewType.TopConnectedCircles,
            Title = "Setup Application",
            Description = "Configure your application settings",
            LogoPath = "Resources/company-logo.svg"
        };

        wizard.NodeChanging += Wizard_NodeChanging;
        wizard.NodeChanged += Wizard_NodeChanged;
        wizard.FinishRequested += Wizard_FinishRequested;
        wizard.CancelRequested += Wizard_CancelRequested;

        this.Controls.Add(wizard);
    }

    private void SetupWizardSteps()
    {
        var steps = new IWizardNode[]
        {
            new WelcomeStep(wizardData),
            new DatabaseConnectionStep(wizardData),
            new UserAccountStep(wizardData),
            new ApplicationSettingsStep(wizardData),
            new ReviewStep(wizardData),
            new CompletionStep(wizardData)
        };

        wizard.AddSteps(steps);
    }

    private void Wizard_NodeChanging(object sender, NodeChangeEventArgs e)
    {
        // Validate current step before allowing navigation
        if (!e.CurrentNode.CanMoveNext && e.ToNode != null)
        {
            var stepValidation = ValidateStep(e.CurrentNode);
            if (!stepValidation.IsValid)
            {
                MessageBox.Show(stepValidation.ErrorMessage, "Validation Error", 
                              MessageBoxButtons.OK, MessageBoxIcon.Warning);
                e.Cancel = true;
            }
        }
    }

    private void Wizard_NodeChanged(object sender, NodeChangeEventArgs e)
    {
        // Save progress and update UI
        SaveWizardProgress();
        UpdateWindowTitle(e.ToNode.Name);
    }

    private void Wizard_FinishRequested(object sender, EventArgs e)
    {
        try
        {
            // Perform final setup operations
            ApplyConfiguration();
            
            MessageBox.Show("Application setup completed successfully!", 
                          "Setup Complete", MessageBoxButtons.OK, MessageBoxIcon.Information);
            
            this.DialogResult = DialogResult.OK;
            this.Close();
        }
        catch (Exception ex)
        {
            MessageBox.Show($"Setup failed: {ex.Message}", "Setup Error", 
                          MessageBoxButtons.OK, MessageBoxIcon.Error);
        }
    }

    private void Wizard_CancelRequested(object sender, EventArgs e)
    {
        var result = MessageBox.Show(
            "Are you sure you want to cancel the setup? All progress will be lost.",
            "Cancel Setup", MessageBoxButtons.YesNo, MessageBoxIcon.Question);

        if (result == DialogResult.Yes)
        {
            this.DialogResult = DialogResult.Cancel;
            this.Close();
        }
    }

    private ValidationResult ValidateStep(IWizardNode step)
    {
        // Implement step-specific validation
        return step switch
        {
            DatabaseConnectionStep dbStep => ValidateDatabaseConnection(dbStep),
            UserAccountStep userStep => ValidateUserAccount(userStep),
            ApplicationSettingsStep settingsStep => ValidateSettings(settingsStep),
            _ => new ValidationResult { IsValid = true }
        };
    }

    private void ApplyConfiguration()
    {
        // Apply all configuration settings
        ConfigurationManager.Instance.ApplySettings(wizardData);
        DatabaseManager.Instance.InitializeConnection(wizardData.DatabaseConfig);
        UserManager.Instance.CreateInitialUser(wizardData.UserConfig);
    }

    private void SaveWizardProgress()
    {
        // Save current progress to allow resumption
        var progress = new WizardProgress
        {
            CurrentStep = wizard.CurrentIdx,
            Data = wizardData,
            Timestamp = DateTime.Now
        };
        ProgressManager.SaveProgress(progress);
    }

    private void UpdateWindowTitle(string stepName)
    {
        this.Text = $"Application Setup Wizard - {stepName}";
    }
}

public class WizardData
{
    public DatabaseConfig DatabaseConfig { get; set; } = new();
    public UserConfig UserConfig { get; set; } = new();
    public ApplicationConfig ApplicationConfig { get; set; } = new();
}

public class ValidationResult
{
    public bool IsValid { get; set; }
    public string ErrorMessage { get; set; }
}

Theming

BeepWizard provides comprehensive theming with automatic application to all panels, buttons, and navigation elements:

Theme Application

var wizard = new BeepWizard
{
    Theme = "DarkTheme",
    ViewType = WizardViewType.TopBreadcrumb
};

// Theme automatically applies to:
// - AppBarTitleBackColor: Top panel background
// - SideMenuBackColor: Side panel and footer backgrounds
// - CardBackColor: Content panel background
// - Button themes: Navigation buttons
// - Stepper/breadcrumb themes: Progress indicators

// Manual theme application
wizard.ApplyTheme();

Wizard Theme Properties

  • AppBarTitleBackColor - Top panel background color
  • SideMenuBackColor - Side panel and footer backgrounds
  • CardBackColor - Main content area background
  • ButtonBackColor - Navigation button backgrounds
  • ButtonForeColor - Navigation button text colors
  • BorderColor - Panel borders and separators

Best Practices

üéØ Wizard Design

  • Clear progression: Design logical step sequences that users can easily follow
  • Validation strategy: Implement comprehensive validation at each step
  • Progress visibility: Choose view types that clearly show wizard progress
  • Save progress: Allow users to save and resume wizard sessions

üì± Layout Selection

  • TopConnectedCircles: Best for 3-7 steps with clear visual progression
  • TopBreadcrumb: Ideal for simple, linear workflows
  • SideList: Perfect for complex wizards with many steps or detailed content
  • Content sizing: Ensure adequate content area for each step's requirements

üîÑ Navigation Management

  • Step validation: Always validate before allowing forward navigation
  • Graceful degradation: Handle validation failures with clear error messages
  • Back navigation: Allow users to go back and modify previous steps
  • Cancel handling: Implement proper cancel confirmation and cleanup

‚ö° Performance

  • Lazy loading: Load step content only when needed
  • Animation efficiency: Use built-in slide animations for smooth transitions
  • Memory management: Dispose of step resources properly
  • Data persistence: Implement efficient data storage between steps
üí° Design Tip

Use TopConnectedCircles for setup wizards, TopBreadcrumb for simple forms, and SideList for complex configuration workflows with multiple sections.

‚úÖ Validation Best Practice

Implement both real-time validation (as users type) and step validation (before navigation) for the best user experience.

‚ö†Ô∏è Navigation Warning

Always handle the NodeChanging event to implement proper validation - this prevents users from proceeding with incomplete or invalid data.
